/*leetcode-300-Longest Increasing Subsequence*/
	 int lengthOfLIS(vector<int>& nums)
	 {
		 int n = nums.size(),ret = 0;
		 if (n == 0) return 0;
		 vector<int>dp(n, 1);
		 for (int i = 0; i < n;i++)
		 {
			 for (int j = 0; j < i; j++)
			 {
				 if(nums[i]>nums[j])dp[i] = max(dp[i], dp[j] + 1);				 
			 }
			 ret = max(ret, dp[i]);
		 }
		 return ret;
		 
	 }
